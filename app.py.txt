import streamlit as st

# Configuração da página do navegador
st.set_page_config(page_title="Protocolo Digital", page_icon="📋", layout="centered")

st.title("📋 Sistema de Registro de Protocolo")
st.markdown("Preencha os campos abaixo para registrar o recebimento dos volumes.")
st.write("---")

# 1. Informações básicas (Campos obrigatórios)
emissor = st.text_input("Nome do funcionário (Emissor do relatório):", placeholder="Ex: João Silva")
destinatario = st.text_input("Destinatário da remessa:", placeholder="Ex: Setor de TI / Maria Souza")
ref_volume = st.text_input("Referência do volume:", placeholder="Ex: Caixa 01, Envelope Pardo, Palete A")

# 2. Quantidade de itens
qtd_itens = st.number_input("Quantidade de itens dentro do volume:", min_value=1, value=1, step=1)

# 3. Nome do Remetente de cada um dos itens (Gerado dinamicamente)
st.subheader("📦 Remetentes por Item")
itens_remetentes = []

for i in range(1, int(qtd_itens) + 1):
    remetente = st.text_input(f"Nome do Remetente do Item {i}:", key=f"item_{i}", placeholder=f"Remetente do item {i}")
    itens_remetentes.append(remetente)

st.write("---")

# 4. Botão para processar e gerar o protocolo
if st.button("Gerar e Salvar Protocolo", type="primary"):
    # Validação dos campos obrigatórios
    if not emissor or not destinatario or not ref_volume or any(not r.strip() for r in itens_remetentes):
        st.error("⚠️ Erro: Todos os campos são obrigatórios! Por favor, preencha as informações do volume e de todos os remetentes.")
    else:
        st.success("✅ Protocolo gerado com sucesso!")
        
        # Montagem do relatório textual
        relatorio = []
        relatorio.append("=============================================")
        relatorio.append("         RELATÓRIO DE RECEBIMENTO          ")
        relatorio.append("=============================================")
        relatorio.append(f" Emitido por:         {emissor}")
        relatorio.append(f" Destinatário:        {destinatario}")
        relatorio.append(f" Ref. do Volume:      {ref_volume}")
        relatorio.append(f" Total de Itens:      {int(qtd_itens)}")
        relatorio.append("---------------------------------------------")
        relatorio.append(" LISTAGEM DE REMETENTES POR ITEM:")
        for idx, reme in enumerate(itens_remetentes, 1):
            relatorio.append(f"   • Item {idx:02d}: Remetente -> {reme}")
        relatorio.append("=============================================")
        
        texto_final = "\n".join(relatorio)
        
        # Exibe o relatório na tela dentro de uma caixa de texto limpa
        st.text_area("Visualização do Protocolo:", value=texto_final, height=250)
        
        # Cria o botão de download para o usuário baixar o arquivo .txt
        st.download_button(
            label="📥 Baixar Relatório em TXT",
            data=texto_final,
            file_name=f"protocolo_{ref_volume.replace(' ', '_')}.txt",
            mime="text/plain"
        )
